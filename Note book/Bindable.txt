class Graphics
{
private:
	wrl::ComPtr<ID3D11Device> mDevice;
	wrl::ComPtr<IDXGISwapChain> mSwapChain;
	wrl::ComPtr<ID3D11DeviceContext> mContext;
	wrl::ComPtr<ID3D11RenderTargetView> mRenderTargetView;
	DxgiInfoManager infoManager;
}

// 可绘制的实体
// 在绘制的函数中，调用所有的bind对象将参数绑定到渲染管线
class Drawable							
{
public:
	void draw(Graphics gfx)
	{
		for (Binable* bind : mBinds)
			bind->bind(gfx);
	}
	void update() = 0;

private:
	QList<Bindable*> mBinds;
}

class Bindable
{
public:
	void bind(Graphics gfx) = 0;		// 接收参数并绑定到渲染管线
}

class VertexBuffer : public Bindable
{
};

class IndexVertexBuffer : public Bindable
{
};

class PixelBuffer : public Bindable
{
};

class ConstBuffer : public Bindable
{
};

class VertexConstBuffer : public ConstBuffer
{
};

class PixelConstBuffer : public ConstBuffer
{
};


class TransformationConstBuffer : public VertexConstBuffer
{
};
